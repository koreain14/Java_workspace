create table movie(
movie_id number primary key
, title varchar2(50)
, genre_id number
, showtime number
, open_day date
);

create sequence seq_movie
increment by 1
start with 1;

create table genre(
genre_id number primary key
, title varchar2(20)
);

create sequence seq_genre
increment by 1
start with 1;


create table score(
score_id number primary key
, star number default 0
, regdate date default sytdate
, movie_id number
);

create sequence seq_score
increment by 1
start with 1;

create table director(
director_id number primary key
, name varchar2(50)
);

create sequence seq_director
increment by 1
start with 1;


create table casting(
cating_id number primary key
, actor varchar2(50)
, movie_id number
);

create sequence seq_casting
increment by 1
start with 1;

select * from movie m, genre g where g.genre_id = m.genre_id;
